CREATE TYPE profile_type_enum AS ENUM (
    'Admin',
    'User',
    'Maker',
    'Checker'
);
CREATE EXTENSION IF NOT EXISTS pgcrypto;

CREATE TABLE create_merchants (

    username VARCHAR(20) NOT NULL UNIQUE,
    password VARCHAR(30) NOT NULL,
    email_id VARCHAR(50) NOT NULL,

    -- BLOCK 1
    pay_in_fees DECIMAL(10,2) DEFAULT 0.00,
    pay_out_fees DECIMAL(10,2) DEFAULT 0.00,

    -- BLOCK 2
    pay_in_code VARCHAR(3) NOT NULL,
    pay_out_code VARCHAR(3) NOT NULL,
    settlement_code VARCHAR(3) NOT NULL,

    -- BLOCK 3
    profile_type profile_type_enum NOT NULL,

    -- BLOCK 4
    created_date DATE NOT NULL,
    created_time TIME NOT NULL,

    -- BLOCK 5
    created_by VARCHAR(20) NOT NULL
);
